export const getShadesOf = (hexColor) => {
  const alphaValues = ["D9", "A6", "73", "40"];
  return [hexColor].concat(alphaValues.map((value) => hexColor + value));
};

export const colors = [
  {
    title: "Primary",
    colors: getShadesOf("#FF8A00"),
  },
  {
    title: "Secondary",
    colors: getShadesOf("#EE6E55"),
  },
  {
    title: "Tertiary",
    colors: getShadesOf("#2BC3E8"),
  },

    title: "Surface 1",

  },

    title: "Surface 2",
    colors: getShadesOf("#DADCDF"),
